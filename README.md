# essential
1. fonts you need to have in order to view everything right.
2. a backup for fonts

# install
copy the fonts to ~/.fonts dir (if dir is not there create one)
$ cd ~/.fonts
$ mkfontscale
$ mkfontdir
$ ttmkfdir -o fonts.dir
$ fc-cache -f -v ~/.fonts
$ fc-cache -f -v
$ fc-match NonExistingFont
